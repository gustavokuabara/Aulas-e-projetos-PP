#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main()
{
    char frase[1000];
    int i, j;
    printf("Digite uma frase: \n");
    fgets(frase, 1000, stdin);
    for(i = 0; i < 999; i++){
        if(frase[i] == 'r' || frase[i] == 'R'){
            if(frase[i + 1] != ' ' && frase[i + 1] != 'r' && frase[i + 1] != 'R' && frase[i + 1] != '\0' && frase[i + 2] != ' ' && frase[i + 2] != '\0' && frase[i - 1] != 'r' && frase[i - 1] != 'R'){
                if(frase[i] == 'r'){
                    frase[i] = 'l';
                }else if(frase[i] == 'R'){
                    frase[i] = 'L';
                }
            }else if((frase[i + 1] == 'r' || frase[i + 1] == 'R') && frase[i + 2] != ' ' && frase[i + 2] != '\0' && frase[i + 1] != ' ' && frase[i + 1] != '\0' && frase[i + 1] != '\n' && frase[i + 2] != '\n'){
                if(frase[i] == 'r'){
                    frase[i] = 'l';
                    frase[i + 1] = 'ç';
                    i++;
                }else if(frase[i] == 'R'){
                    frase[i] = 'L';
                    frase[i + 1] = 'ç';
                    i++;
                }
            }else if(frase[i + 2] == ' '){
                if(frase[i + 1] != 'r' && frase[i + 1] != 'R'){
                    if(frase[i] == 'r'){
                        frase[i] = 'l';
                    }else if(frase[i] == 'R'){
                        frase[i] = 'L';
                    }
                }
            }
        }
    }
    printf("Frase convertida: \n");
    for(i = 0; frase[i] != '\0';i++){
        if(frase[i] != 'ç'){
            printf("%c",frase[i]);
        }
    }
    system("pause");
    return 0;
}
